Java docs - /** */: 

This is documentation that is inserted inside your java file. It explains the structure of 
your coding. The syntax is very similar to html tagging.

The sytax is as follows:
@<tag-name> <text>

Class 		-- C
Interface 	-- I
Method		-- M
Variable	-- V

TagName		UsedWith	Tag Description
@author		C, Z		
@deprecated	C, I, M, V
@exception	M
{@link}		C, I, M, V	To create links within the doc
@param		M		Methods that take in parameters
@return		M
@see		C, I, M, V	Also like link, but link has underline/italic, but see does not
@since		C, I, M, V	When was this created
@throws		M		same as @exception
@version	C, I


Example:
@see <class name or interface name> <label>
@see <full package name> <label>
{@link <class name or interface name> <label>}
{@link <full package name>}

@see com.example.extras.Stack stack
see also this {@link java.util.Stack Stack} class
For detailed information, try{@link com.example.extras.Stack}

Stack.java

// Filename: Stack.java
package com.example.extras.util;

/**
 *  This class implements a generic stack. An array is
 *  used to implement the stack.
 *  @author K.A.M.
 *  @version 1.1
 */
public class Stack {

    /** The array that implements the stack. */
    private Object[] stackArray;

    /** The top of the stack. */
    private int topOfStack;

    /**
     *  Initialize the stack.
     *  @param capacity Length of the stack.
     */
    public Stack (int capacity) {
        stackArray = new Object[capacity];
        topOfStack = -1;
    }

    /**
     *  Push a value on the stack.
     *  @param element The object to push on the stack.
     *  @exception FullStackException The stack is full.
     *  @see #pop()
     */
    public synchronized void push(Object element) throws FullStackException {
        if (isFull()) throw new FullStackException();
        stackArray[++topOfStack] = element;
    }

    /**
     *  Insert a value on the stack.
     *  @param element The object to insert on the stack.
     *  @exception FullStackException The stack is full.
     *  @deprecated As of version 1.1,
     *  replaced by the {@link #push(Object) push} method. The insert name
     *  was misleading.
     */
    public synchronized void insert(Object element) throws FullStackException {
        push(element);
    }

    /**
     *  Pop the value from top of the stack.
     *  @return The object on top of the stack.
     *  @exception EmptyStackException The stack is empty.
     *  @see #push(Object)
     */
    public synchronized Object pop() throws EmptyStackException {
        if (isEmpty()) throw new EmptyStackException();
        Object obj = stackArray[topOfStack];
        stackArray[topOfStack] = null;
        topOfStack--;
        return obj;
    }

    /**
     *  Peek at the object on top of the stack. The stack is
     *  not popped.
     *  @return The object on top of the stack.
     *  @exception EmptyStackException The stack is empty.
     */
    public synchronized Object peek() throws EmptyStackException {
        if (isEmpty()) throw new EmptyStackException();
        return stackArray[topOfStack];
    }

    /**
     *  Check if the stack is empty.
     *  @return true, if stack is empty.
     */
    public boolean isEmpty() { return topOfStack < 0; }

    /**
     *  Check if the stack is full.
     *  @return true, if stack is full.
     */
    public boolean isFull() {return topOfStack == stackArray.length - 1; }
}


To create the docs above use the following commandline:
javadoc -author -version -private *.java
